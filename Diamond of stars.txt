/* Print the following pattern for the given number of rows.
Note: N is always odd.
Pattern for N = 5
		..*
		.***
		*****
		.***
		..*
The dots represent spaces.
*/
#include<iostream>
using namespace std;

int main() {
    int n;
    cout<<"Enter n";
    cin >> n;
    int i = 1;
    while(i <= (n/2) + 1) {
        int k = 1; 
        while(k <= (n/2)-i+1) {
            cout << " ";
			k++;
        }
        int j = 1;
        while(j <= (2*i) - 1) {
            cout << "*";
            j++;
        }
        cout << endl;
        i++;
    }
    i = 1;
    while(i <= n / 2){
        int k = 1;
        while(k <= i) {
            cout << " ";
            k++;
        }
        int j = 2 * ((n/2) - i + 1) - 1;
        while(j >= 1) {
            cout << "*";
            j--;
        }
        cout << endl;
        i++;
    }
}
